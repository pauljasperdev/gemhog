    1. Create `apps/web/src/lib/analytics.ts` -- a thin wrapper around posthog.capture():
       ```typescript
       import posthog from "posthog-js";

       export const AnalyticsEvents = {
         LANDING_PAGE_VIEWED: "landing_page_viewed",
         SIGNUP_STARTED: "signup_started",
         SIGNUP_COMPLETED: "signup_completed",
       } as const;

       export function trackEvent(
         event: (typeof AnalyticsEvents)[keyof typeof AnalyticsEvents],
         properties?: Record<string, unknown>,
       ) {
         posthog.capture(event, {
           referrer: typeof document !== "undefined" ? document.referrer : undefined,
           ...properties,
         });
       }
       ```

       This module:
       - Exports typed event name constants (snake_case per context decision)
       - Attaches `document.referrer` to all events for channel attribution
       - UTM parameters are auto-captured by PostHog SDK from URL
       - Does NOT call posthog.identify() (anonymous only)

    2. Add `landing_page_viewed` event to `apps/web/src/app/page.tsx`:
       - Import `trackEvent` and `AnalyticsEvents` from `@/lib/analytics`
       - Fire `trackEvent(AnalyticsEvents.LANDING_PAGE_VIEWED)` in a `useEffect`
         on mount (empty dependency array). This distinguishes landing page views
         from other auto-captured pageviews.
       - The page is already "use client" so no changes needed there.

    3. Create `apps/web/src/app/verify/verify-analytics.tsx` -- a small client
       component that fires `signup_completed` on mount when verification succeeds:
       ```typescript
       "use client";

       import { useEffect } from "react";
       import { trackEvent, AnalyticsEvents } from "@/lib/analytics";

       export function VerifyAnalytics({ status }: { status: string }) {
         useEffect(() => {
           if (status === "success") {
             trackEvent(AnalyticsEvents.SIGNUP_COMPLETED);
           }
         }, [status]);

         return null;
       }
       ```
